## 开盘时运行函数
def market_open(context):
    if g.security not in context.portfolio.positions:
        orders =order(g.security,1000) 
        print(orders) 
        
        if orders is None:
            print("创建订单失败...") 
        else:
            print("""交易费用：{}""".format(orders.commission))
            print("""是否买单：{}""".format(orders.is_buy))
            print("""订单状态：{}""".format(orders.status))
            print("""订单平均成交价格：{}""".format(orders.price))
    else:
        order(g.security,-800)
        
    print(type(context.portfolio.long_positions))
    long_positions_dict =context.portfolio.long_positions
    for position in list(long_positions_dict.value()):
        print("""标的：{}。总仓位：{}。标的价值：{}。建仓时间：{}。""".format(position.security,position.total_amount,position.value,position.init_time))
 
## 收盘后运行函数  
def after_market_close(context):
    print("闭市后")
    trades=get_trades()
    for _trade in trades.values():
        print("""成交纪录：{}""".format(_trade))
        print("""成交时间：{}""".format(_trade.time))
        print("""对应订单号：{}""".format(_trade.order_id))
